模块化方案  
```
// 函数作用域：在全局访问不了 obj.bar 和 obj.bar
// 只做到了隔离，没有做到相互的通信。  
function foo() {
  var bar = 1;
  var bar = 2;
}
```
```
var obj = {};
function resolve(obj) {
  obj.bar = 1;
  obj.bar = 2;
}
ewsolve(obj)
```
```
// comonJS 规范雏形
var module = {
  exports: {}
}
function foo(module, exports) {
  exports.bar = 1;
  exports.bar = 2;
}
foo(module, module.exports)
```
